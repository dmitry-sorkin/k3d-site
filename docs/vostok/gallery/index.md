Этот принтер стал чем-то новым и не похожим на прошлые проекты! С механикой и корпусом всё было интуитивно понятно, а вот прошивка заставила меня как следует погрузиться в Klipper. Детали оси Х и головы - PA-GF30, остальные внутренние детали PP-GF30, наружные - ASA-CF.<br>Искренняя благодарность Дмитрию за прекрасно проработанный проект - никаких существенных проблем при сборке не возникло. // Переключение постов по якорю в URL. Показывается либо пост, заголовок
// которого указан в hash, либо первый пост при открытии страницы без якоря.
(function () {
  function update() {
    var posts = document.querySelectorAll('.gallery-post');
    if (!posts.length) return;
    var id = window.location.hash.replace('#', '');
    var target = id ? document.getElementById(id) : null;
    var active = target ? target.closest('.gallery-post') : null;
    for (var i = 0; i < posts.length; i++) {
      posts[i].classList.toggle('active', posts[i] === active);
    }
    if (!active && posts[0]) posts[0].classList.add('active');
  }
  document.addEventListener('DOMContentLoaded', update);
  window.addEventListener('hashchange', update);
})();
